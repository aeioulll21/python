print("Create a character!")
name = input("What is your character called? ")
age = input(f"How old is {name}? ")
strengths = input(f"What are {name}'s strengths? ")
weaknesses = input(f"What are {name}'s weaknesses? ")
print(f"Your character's name is {name}.")
print(f"{name} is {age} years old.")
print(f"Strengths: {strengths}.")
print(f"Weaknesses: {weaknesses}.")
print(f"{name} says, 'Thanks for creating me!'")
